// BallSim Main Script
const simulation = new BallSim.Simulation({
  gravity: 9.8,
  friction: 0.1,
  width: 800,
  height: 600
});

// Create a container (walls)
const container = simulation.createContainer({
  type: 'rectangle',
  width: 800,
  height: 600,
  elasticity: 0.8
});

// Create balls
for (let i = 0; i < 50; i++) {
  simulation.createBall({
    radius: 10 + Math.random() * 20,
    mass: 1,
    position: {
      x: Math.random() * 700 + 50,
      y: Math.random() * 500 + 50
    },
    velocity: {
      x: Math.random() * 10 - 5,
      y: Math.random() * 10 - 5
    },
    color: `hsl(${Math.random() * 360}, 80%, 60%)`
  });
}

simulation.start();

Quick Documentation

BallSim.Simulation

Creates a new simulation instance with the specified parameters.

Parameters:
  • gravity (Number): Gravity force value (default: 9.8)
  • friction (Number): Global friction coefficient (default: 0.1)
  • width (Number): Width of simulation area
  • height (Number): Height of simulation area
